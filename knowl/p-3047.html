\begin{equation*} A_3 = \left[\begin{array}{cc} 0 \amp 0\cr 1 \amp 0 \end{array}\right], \ A_4 = \left[\begin{array}{cc} 0 \amp 0\cr 0 \amp 1 \end{array}\right], \end{equation*}
